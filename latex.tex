    \section{Table}\label{sec:table}
		\subsection{Muliline Cells}
			\ldots with package \textit{makecell}
			
			\begin{tabular}{cl}
				A		& B  \\
				\hline
				C		& \makecell[tl]{D \\ E \\ F} \\
				\hline
				G		& \makecell[tr]{H \\ I}
		    \end{tabular}
	    
	    \subsection{In enumerate env}
		    	\ldots without text in first line.
			\newcommand{\makeTable}[0]{
				\begin{tabular}{l | l}
					A & B \\
					C & D \\
					E & F
				\end{tabular}				
			}	    
		    
		    \begin{enumerate}
		    		\item
		    			\makeTable
	
		    			This sucks :(
		    			
		    		\item
		    			\adjustbox{valign=t}{
		    				\makeTable
		    			}
	
		    			This is better :|
	    			
		    			
		    		\item
					\begin{adjustbox}{valign=t, tabular=l|l}
						A & B \\
						C & D \\
						E & F
					\end{adjustbox}
					
					This is great :)
		    \end{enumerate}
